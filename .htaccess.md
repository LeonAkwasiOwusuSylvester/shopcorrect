RewriteEngine On

# 1. Direct shopcorrect.com/store/any-slug to public/store.php?slug=any-slug
RewriteRule ^store/([a-zA-Z0-9-]+)$ public/store.php?slug=$1 [L,QSA]

# 2. Optional: Clean up other public URLs (removes .php extension)
RewriteCond %{REQUEST_FILENAME} !-f
RewriteCond %{REQUEST_FILENAME} !-d
RewriteRule ^([^\.]+)$ public/$1.php [NC,L]
